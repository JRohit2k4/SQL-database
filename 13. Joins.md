## JOINS

-- 1. INNER JOIN  
only matching data from both table  
`SELECT c.name, o.order_id
FROM customers c
INNER JOIN orders o
ON c.id = o.customer_id;`

-- 2. LEFT JOIN  
all data from left table and only matching for right table  
`SELECT c.name, o.order_id
FROM customers c
LEFT JOIN orders o
ON c.id = o.customer_id;`

-- 3. Customers with no orders  
`SELECT c.name
FROM customers c
LEFT JOIN orders o
ON c.id = o.customer_id
WHERE o.order_id IS NULL;`

-- 4. Total orders per customer  
`SELECT c.name, COUNT(o.order_id) AS total_orders
FROM customers c
LEFT JOIN orders o
ON c.id = o.customer_id
GROUP BY c.name;`

-- 5. SELF JOIN (employee & manager)  
`SELECT e.name AS employee, m.name AS manager
FROM employee e
LEFT JOIN employee m
ON e.manager_id = m.id;`

-- 6. Join + aggregation  
`SELECT c.name, SUM(o.amount) AS total_spent
FROM customers c
JOIN orders o
ON c.id = o.customer_id
GROUP BY c.name;`
