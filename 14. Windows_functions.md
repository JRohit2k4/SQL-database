-- WINDOW FUNCTIONS

-- Sample Table: orders

  
| customer_id| amount | city |
|---|---|---|
| 1          | 500    | Pune |
| 2          | 700    | Mumbai |
| 1          | 300    | Pune |
| 3          | 900    | Delhi |
--------------------------------------------------

-- 1. ROW_NUMBER
-- Assigns unique number to each row

SELECT customer_id, amount,
ROW_NUMBER() OVER (ORDER BY amount DESC) AS rn
FROM orders;

-- Output:

| customer_id| amount | rn |
|---|---|---|
| 3          | 900    | 1  |
| 2          | 700    | 2  |
| 1          | 500    | 3  |
| 1          | 300    | 4  |

--------------------------------------------------

-- 2. RANK
-- Same rank for ties, skips next rank

SELECT customer_id, SUM(amount) AS total,
RANK() OVER (ORDER BY SUM(amount) DESC) AS rnk
FROM orders
GROUP BY customer_id;

-- Output:

| customer_id| total | rnk |
|---|---|---|
| 3          | 900   | 1   |
| 2          | 700   | 2   |
| 1          | 800   | 3   |

--------------------------------------------------

-- 3. DENSE_RANK
-- Same rank for ties, no skipping

SELECT customer_id, SUM(amount) AS total,
DENSE_RANK() OVER (ORDER BY SUM(amount) DESC) AS drnk
FROM orders
GROUP BY customer_id;

-- Output:
|customer_id| total | drnk |
|---|---|---|
| 3          | 900   | 1    |
| 1          | 800   | 2    |
| 2          | 700   | 3    |

--------------------------------------------------

-- 4. Running Total
-- Cumulative sum

SELECT customer_id, amount,
SUM(amount) OVER (ORDER BY customer_id) AS running_total
FROM orders;

-- Output:
| customer_id| amount | running_total |
|---|---|---|
| 1          | 500    | 500           |
| 1          | 300    | 800           |
| 2          | 700    | 1500          |
| 3          | 900    | 2400          |

--------------------------------------------------

-- 5. PARTITION BY
-- Total per city

SELECT city, customer_id, amount,
SUM(amount) OVER (PARTITION BY city) AS city_total
FROM orders;

-- Output:
| city   | customer_id| amount | city_total |
|---|---|---|---|
| Pune   | 1          | 500    | 800        |
| Pune   | 1          | 300    | 800        |
| Mumbai | 2          | 700    | 700        |
| Delhi  | 3          | 900    | 900        |

--------------------------------------------------

-- 6. LAG
-- Previous row value

SELECT customer_id, amount,
LAG(amount) OVER (ORDER BY customer_id) AS prev_amount
FROM orders;

-- Output:
| customer_id| amount | prev_amount |
|---|---|---|
| 1          | 500    | NULL        |
| 1          | 300    | 500         |
| 2          | 700    | 300         |
| 3          | 900    | 700         |

--------------------------------------------------

-- 7. Top 2 customers
-- Based on total purchase

SELECT customer_id, total
FROM (
    SELECT customer_id, SUM(amount) AS total,
    RANK() OVER (ORDER BY SUM(amount) DESC) AS rnk
    FROM orders
    GROUP BY customer_id
) t
WHERE rnk <= 2;

-- Output:
| customer_id| total |
|---|---|
| 3          | 900   |
| 1          | 800   |
